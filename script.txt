let cart = [];
let total = 0;
let orderId = "";

function addToCart(name, price) {
  cart.push({name, price});
  total += price;
  renderCart();
}

function renderCart() {
  let list = document.getElementById("cartItems");
  list.innerHTML = "";
  cart.forEach(item => {
    let li = document.createElement("li");
    li.textContent = item.name + " - ‚Çπ" + item.price;
    list.appendChild(li);
  });
  document.getElementById("total").innerText = total;
}

function placeOrder() {
  if(cart.length === 0) {
    alert("Cart is empty");
    return;
  }
  orderId = "VR" + Math.floor(Math.random()*100000);
  document.getElementById("orderMsg").innerText =
    "‚úÖ Order Placed Successfully! Your Order ID: " + orderId;
  cart = [];
  total = 0;
  renderCart();
}

function trackOrder() {
  let id = document.getElementById("trackId").value;
  if(id === orderId) {
    document.getElementById("trackResult").innerText =
      "üì¶ Your order is being packed & will be shipped soon.";
  } else {
    document.getElementById("trackResult").innerText =
      "‚ùå Invalid Order ID";
  }
}
